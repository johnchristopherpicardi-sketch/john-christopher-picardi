// Simple demo: shows placeholders for your paintings

const inventory = {
  "example_post_1": { price: "$1200", status: "available" },
  "example_post_2": { status: "sold" }
};

const container = document.getElementById("instafeed");

Object.keys(inventory).forEach(key => {
  const item = inventory[key];
  const div = document.createElement("div");
  div.className = "painting";
  div.style.height = "200px";
  div.style.background = "#ccc";
  div.style.position = "relative";
  div.textContent = key;

  if(item.status === "sold"){
    const sold = document.createElement("div");
    sold.className = "sold-label";
    sold.textContent = "Sold";
    div.appendChild(sold);
  } else if(item.price){
    const price = document.createElement("div");
    price.className = "price-label";
    price.textContent = item.price;
    div.appendChild(price);
  }

  container.appendChild(div);
});
